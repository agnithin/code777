  7 /** Creates an View for Answers
  8     @class Represents a player in Code 777 Game
  9       
 10     @property {Game} game the model game.
 11     @property {HTMLElement} element generated HTML element.
 12 */
 13 
 14 function AnswerView(_game){
 15 	this.game = _game;
 16 	this.element = document.createElement('table');
 17 	
 18 	this.draw(this.game.answeredQuestion)
 19 }
 20 
 21 /**
 22  * generate HTML element for the list of answers
 23  * @param _aQuestionList list of questions
 24  */
 25 AnswerView.prototype.draw = function(_aQuestionList){
 26 	for(var i=0; i<_aQuestionList.length; i++){
 27 		this.drawQuestion(_aQuestionList[i]);
 28 	}
 29 	
 30 }
 31 
 32 /**
 33  * adds a row with the question information
 34  * @param _aQuestion details of the answered question
 35  */
 36 AnswerView.prototype.drawQuestion = function(_aQuestion){	
 37 	
 38 	var answerElement = document.createElement('tr');
 39 	answerElement.setAttribute('class', 'player_answer');
 40 	
 41 	var aPlayerNameElement = document.createElement('td');
 42 	aPlayerNameElement.setAttribute('class', 'a_player_name');
 43 	aPlayerNameElement.appendChild(document.createTextNode(_aQuestion.player.playerName));
 44 	answerElement.appendChild(aPlayerNameElement);
 45 	
 46 	var aQuestionElement = document.createElement('td');
 47 	aQuestionElement.setAttribute('class', 'a_question');
 48 	aQuestionElement.appendChild(document.createTextNode(_aQuestion.question.questionText));
 49 	answerElement.appendChild(aQuestionElement);
 50 	
 51 	var aAnswerElement = document.createElement('td');
 52 	aAnswerElement.setAttribute('class', 'a_answer');
 53 	aAnswerElement.appendChild(document.createTextNode(_aQuestion.answer));
 54 	answerElement.appendChild(aAnswerElement);
 55 	
 56 	this.element.appendChild(answerElement);
 57 }
