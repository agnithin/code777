  7 /** Creates an model for Game
  8     @class Represents a Code 777 Game
  9       
 10     @property {Deck} deck Deck of cards for the game.
 11     @property {Number} currentPlayer player id of the player whose turn it is.
 12     @property {Number} playerId player id of the player whose is playing the game.
 13     @property {Object} players associative array of players of the game.
 14     @property {Number[]} questionOrder shuffled order of questions to ask.
 15     @property {AnsweredQuestion[]} answeredQuestion array of AnsweredQuestion objects; contains information of the answered questions.
 16     @property {Boolean} gameMode if the game is single player or multiplayer
 17 */
 18 function Game(_gameMode, _playerId){
 19 	this.deck = new Deck();
 20 	this.currentPlayer;// player id of the player whose turn it is;
 21 	this.playerId = _playerId; 
 22 	this.players = new Object();
 23 	this.questionOrder = new Array();
 24 	this.answeredQuestion = new Array();
 25 	this.gameMode = _gameMode;
 26 	
 27 	this.createNewGame();
 28 	this.dealCards();
 29 	this.initializeQuestionSet();
 30 }

 73 	for(var i in Color){
 74 		colorsSet[i] = 0;
 75 	}		
 76 	for(var i in this.players){
 77 		if(i != this.currentPlayer){
 78 			for(var j=0; j<this.players[i].rack.length; j++){
 79 				colorsSet[this.players[i].rack[j].color] ++;
 80 			}

162 /**
163  * answer a particular question
164  * @param _questionId is the question id of the question to be answered
165  * @return the question that is answered
166  */
167 Game.prototype.answerQuestion = function(_questionId){	
168 	var answer;
169 	
170 	//Caluclating questions is divided into 4 sets
171 	if(_questionId < 7){
172 		answer = this.countRackQuestion(_questionId);
173 	}else if(_questionId < 11){
174 		answer = this.countColorsQuestion(_questionId)
175 	}else if(_questionId < 16){
176 		answer = this.checkCardQuestion(_questionId);
177 	}else{
178 		answer = this.checkColorsQuestion(_questionId);
179 	}
180 	
181 	return new AnsweredQuestion(this.players[this.currentPlayer], new Question(_questionId), answer)
182 	
183 };

185 /**
186  * questions 1 - 7
187  * @param _questionId is the question id of the question to be answered
188  * @return the answer
189  */
190 Game.prototype.countRackQuestion = function(_questionId){
191 	var numOfRacks = 0;
192 	
193 	switch(_questionId){
194 		case 0: //1.how many racks have sum above 18
195 		
196 			for(var i in this.players){
197 				if(i != this.currentPlayer){
198 					if(this.players[i].sumOfrack() >= 18){
199 						numOfRacks++;
200 					}
201 				}
202 			}		
203 			break;
204 			
205 		case 1: //2.how many racks have sum below 12
206 			for(var i in this.players){
207 				if(i != this.currentPlayer){
208 					if(this.players[i].sumOfrack() <= 12){
209 						numOfRacks++;
210 					}
211 				}
212 			}		
213 			break;
214 			
215 		case 2://3.On how many racks is there the same number in different colors
216 			for(var i in this.players){
217 				if(i != this.currentPlayer){
218 					//fix logic
219 					if(((3-this.players[i].numberOfDifferentNumbers()) - (this.players[i].numberOfIdenticalCards()-1)) > 1){
220 						numOfRacks++;
221 					}
222 				}
223 			}		
224 			break;
225 			
226 		case 3: //4.On how many racks are there three different colors?
227 			for(var i in this.players){
228 				if(i != this.currentPlayer){
229 					if(this.players[i].numberOfColors() == 3){
230 						numOfRacks++;
231 					}
232 				}
233 			}
234 			break;
235 			
236 		case 4: //On how many racks are the numbers either all even or all uneven?
237 			for(var i in this.players){
238 				if(i != this.currentPlayer){
239 					if(this.players[i].isAllNumbersEven() || this.players[i].isAllNumbersOdd()){
240 						numOfRacks++;
241 					}
242 				}
243 			}		
244 			break;
245 			
246 		case 5: //On how many racks are there at least two identical cards (same number and same color)?
247 			for(var i in this.players){
248 				if(i != this.currentPlayer){
249 					if(this.players[i].numberOfIdenticalCards() > 1){
250 						numOfRacks++;
251 					}
252 				}
253 			}		
254 			break;
255 			
256 		case 6: //On how many racks do you see three consecutive numbers?
257 			for(var i in this.players){
258 				if(i != this.currentPlayer){
259 					if(this.players[i].isAllNumbersConsecutive()){
260 						numOfRacks++;
261 					}
262 				}
263 			}
264 		
265 			break;
266 	}
267 	return numOfRacks;
268 
269 };

271 /**
272  * Questions 8 - 11
273  * @param _questionId is the question id of the question to be answered
274  * @return the answer
275  */
276 Game.prototype.countColorsQuestion = function(_questionId){
277 	var numOfCol = 0;
278 	switch(_questionId){
279 		case 7: //How many colors do you see?
280 		
281 			var colorSet = this.calculateColors();
282 			for(var i in colorSet){
283 				if(colorSet[i] != 0){
284 					numOfCol ++;
285 				}
286 			}
287 			break;
288 		case 8: //How many colors appear at least three times?
289 			var colorSet = this.calculateColors();
290 			for(var i in colorSet){
291 				if(colorSet[i] >=3){ 
292 					numOfCol ++;
293 				}
294 			}
295 		
296 			break;
297 		case 9: //How many numbers are missing
298 			var numberSet = this.calculateNumbers();
299 			for(var i in numberSet){
300 				if(numberSet[i] == 0){ 
301 					numOfCol ++;
302 				}
303 			}
304 		
305 			break;
306 		case 10: //How many of the following cards do you see: green 1, black 5, pink 7?
307 			numOfCol += this.findNumberOfOccurances(1, Color.green);
308 			numOfCol += this.findNumberOfOccurances(5, Color.black);
309 			numOfCol += this.findNumberOfOccurances(7, Color.pink);
310 			
311 			break;		
312 	}
313 	return numOfCol;
314 
315 };

317 /**
318  * Questions 12 - 16
319  * @param _questionId is the question id of the question to be answered
320  * @return the answer 
321  */
322 Game.prototype.checkCardQuestion = function(_questionId){
323 	var leftValue = 0;
324 	var rightValue = 0;
325 	
326 	switch(_questionId){
327 		case 11: //12. Do you see more 3s or more pink 6s?		
328 			var numSet = this.calculateNumbers();
329 			leftValue = numSet[3];
330 			rightValue = this.findNumberOfOccurances(6, Color.pink);
331 			break;
332 			
333 		case 12: //13. Do you see more green 6s or more yellow 7s? 
334 			leftValue = this.findNumberOfOccurances(6, Color.green);
335 			rightValue = this.findNumberOfOccurances(6, Color.yellow);		
336 			break;
337 			
338 		case 13: //14. Do you see more 2s or more yellow 7s?
339 			var numSet = this.calculateNumbers();
340 			leftValue = numSet[2];
341 			rightValue = this.findNumberOfOccurances(7, Color.yellow);		
342 			break;
343 			
344 		case 14: //15. Do you see more pink 6s or more green 6s?
345 			leftValue = this.findNumberOfOccurances(6, Color.pink);
346 			rightValue = this.findNumberOfOccurances(6, Color.green);
347 			break;
348 			
349 		case 15: //16. Do you see more blue 7s or more 7s of other colors?
350 			leftValue = this.findNumberOfOccurances(7, Color.blue);
351 			var numSet = this.calculateNumbers();
352 			rightValue = numSet[7] - leftValue;
353 			break;	
354 	}
355 	
356 	if(leftValue == rightValue){
357 		return '=';
358 	}else if(leftValue < rightValue){
359 		return '<';
360 	}else{
361 		return '>';
362 	}
363 };
