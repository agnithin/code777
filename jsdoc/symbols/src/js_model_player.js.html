  7 /** Creates an model for Player
  8     @class Represents a player in Code 777 Game
  9       
 10     @property {Number} playerId player id of the player.
 11     @property {String} playerName Player name f the player.
 12     @property {Cards[]} rack set of cards that the player holds 
 13     @property {Number} gamesWon Number of games won my the player; if the count becomes 3 the game is over
 14 */
 15 function Player(_playerId, _playerName){
 16 	this.playerId = _playerId;
 17 	this.playerName = _playerName;
 18 	this.rack = new Array();
 19 	this.gamesWon = 0;
 20 }

 22 /**
 23  * find the number of occurrences of a particular card in the deck
 24  * @param _number number value of the card
 25  * @param _color color value of the card
 26  * @return numCards number of occurances of the card
 27  */
 28 Player.prototype.findCard = function(_number, _color){
 29 	var numCards = 0;
 30 	
 31 	for(var i=0; i<this.rack.length; i++){
 32 		if((this.rack[i].number == _number) && (this.rack[i].color == _color)){
 33 			numCards++
 34 		}
 35 	}
 36 	return numCards;
 37 };

112 /**
113  * Calculate the number of identical cards
114  * @return count of identical cards
115  */
116 Player.prototype.numberOfIdenticalCards = function(){
117 	var identicalCards = 1;
118 	for(var i=0; i<this.rack.length; i++){
119 		for(var j=(i+1); j<this.rack.length; j++)
120 			//console.log(i+":"+j+"|" + this.rack[i].number+this.rack[i].color + "| c2=" + this.rack[j].number + this.rack[j].color);
121 			if((this.rack[i].number == this.rack[j].number) && (this.rack[i].color == this.rack[j].color)){
122 				identicalCards++
123 			}
124 	}
125 	return identicalCards;
126 };

129  * check if the all the cards in the deck are consecutive numbers

131  */
132 Player.prototype.isAllNumbersConsecutive = function(){
133 	var allConsecutive = true;
134 	var numberArray = new Array();
135 	
136 	for(var i=0; i<this.rack.length; i++){
137 		numberArray.push(this.rack[i].number);
138 	}
139 	
140 	numberArray.sort();
141 	
142 	for(var i=0; i<(numberArray.length-1); i++){
143 		if((numberArray[i]+1) != numberArray[i+1] ){
144 			allConsecutive = false;
145 		}
146 	}
147 	return allConsecutive;
