  7 /** Creates an View for the deck of cards; used as a rough sketch of cards that can be used by the player
  8     @class Represents a view of deck
  9       
 10     @property {Deck} deck the model deck.
 11     @property {HTMLElement} element generated HTML element.
 12 */
 13 
 14 function DeckView(_deck){
 15 	this.deck = _deck;
 16 	this.element = document.createElement('div');
 17 	
 18 	this.draw();
 19 	
 20 }
 21 
 22 /**
 23  * function generates an HTML element with the deck information
 24  */
 25 DeckView.prototype.draw = function(){
 26 	
 27 	var k=0, t;
 28 	for(var i=1; i<=7 ;i++){
 29 		t = document.createElement("div");
 30 		t.setAttribute("class", "player");
 31 		for(var j=0; j<4; j++){
 32 			var sCard = this.drawCard(this.deck.cards[k++]);
 33 			sCard.onclick = //function(number){ return function a(){alert(number);}}(sCard.number);
 34 			function (sCard){ 
 35 				return function(){$(this).simpledialog({
 36 			    'mode' : 'bool',
 37 			    'prompt' : 'Card ?',
 38 			    'useModal': true,
 39 			    'buttons' : {
 40 			      'Yes': {
 41 			        click: function () {
 42 			          console.log('card selected');
 43 			          sCard.style.fontWeight = 'bold';
 44 			          sCard.style.textDecoration = 'underline';			          
 45 			          sCard.onclick = '';
 46 			        }
 47 			      },
 48 			      'No': {
 49 				        click: function () {
 50 				          console.log('card crossed');
 51 				          sCard.innerHTML= 'X';
 52 				          sCard.style.color = '#000';
 53 				          sCard.style.backgroundColor = '#FFF';
 54 				          sCard.onclick = '';
 55 				        }
 56 				   },
 57 			      'Cancel': {
 58 			        click: function () {
 59 			          console.log('cancelled');
 60 			        },
 61 			        icon: "delete",
 62 			        theme: "b"
 63 			      }
 64 			    }
 65 			  })}
 66 			}(sCard);
 67 			t.appendChild(sCard);				
 68 		}
 69 		this.element.appendChild(t);
 70 	}
 71 }
 72 
 73 /**
 74  * create a new div for a card
 75  * @param _card the details of the card
 76  * @return the HTML element with the details of teh card
 77  */
 78 DeckView.prototype.drawCard = function(_card){
 79 	var element = document.createElement('div');
 80 	element.setAttribute('class', 'card');
 81 	element.style.backgroundColor = _card.color;
 82 	var cardText = document.createTextNode(_card.number);
 83 	element.appendChild(cardText);
 84 	return element;
 85 };
