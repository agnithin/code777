103 /**
104  * function to shuffle an array
105  * @param theArray array to be shuffled
106  */
107 function arrayShuffle(theArray) {
108  	var len = theArray.length;
109 	var i = len;
110 	 while (i--) {
111 	 	var p = parseInt(Math.random()*len);
112 		var t = theArray[i];
113   		theArray[i] = theArray[p];
114 	  	theArray[p] = t;
115  	}
116 };
