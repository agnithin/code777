  7 /** Creates an view for Game
  8     @class Represents a board in Code 777 Game
  9       
 10     @property {Game} game the model game.
 11     @property {HTMLElement} element generated HTML element.
 12 */
 13 
 14 function GameView(_game){
 15 	this.game = _game;
 16 	this.element = document.createElement('div');
 17 	
 18 	for(var i in this.game.players){
 19 		if(i != this.game.playerId){
 20 			this.element.appendChild(this.drawPlayerRack(this.game.players[i]));
 21 		}
 22 	}
 23 }
 24 
 25 /**
 26  * create a div element for rack. place the cards inside the rack
 27  * @param _player details of the player
 28  * @return the HTML element with player details
 29  */
 30 GameView.prototype.drawPlayerRack = function(_player){
 31 	
 32 	var element = document.createElement("div");
 33 	element.setAttribute("class", "player");
 34 	
 35 	var playerLabel = document.createElement("div");
 36 	playerLabel.setAttribute("class", "player_label");
 37 	var playerLabelText = document.createTextNode(_player.playerName);
 38 	playerLabel.appendChild(playerLabelText);
 39 	
 40 	element.appendChild(playerLabel);
 41 	var playerRack = document.createElement("div");
 42 	playerRack.setAttribute("class", "player_rack");
 43 	
 44 	
 45 	for(var i=0; i<_player.rack.length; i++){
 46 		playerRack.appendChild(this.drawCard(_player.rack[i]));
 47 	}
 48 	element.appendChild(playerRack);
 49 	return element;
 50 };
 51 
 52 /**
 53  * create a new div element for a card
 54  * @param _card details of the card
 55  * @returns the HTML element with card details
 56  */
 57 GameView.prototype.drawCard = function(_card){
 58 	var element = document.createElement('div');
 59 	element.setAttribute('class', 'card');
 60 	element.style.backgroundColor = _card.color;
 61 	var cardText = document.createTextNode(_card.number);
 62 	element.appendChild(cardText);
 63 	return element;
 64 };
